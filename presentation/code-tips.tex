qsort :: (Ord a) => [a] -> [a]
qsort [] = []
qsort (x:xs) =
    let leq = [a | a <- xs, a <= x]
        gt = [a | a <- xs, a > x]
        in qsort leq ++ [x] ++ qsort gt
\end{lstlisting}
